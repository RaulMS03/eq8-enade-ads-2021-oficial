O código Java, a seguir, contém a implementação de uma Pilha utilizando a estratégia encadeada.

public class No{
   public int dado;
   public No prox;
   public No(int dado){
       this.dado = dado;
    }
}
public class PilhaEncadeada{
   private No topo;
   public int pop(){
   if (topo == null)
       return -1;
       No lixo = topo;
       topo = topo.prox;
       lixo.prox = null;
       return lixo.dado;
    }
   public int top(){
       if (topo == null)
          return -1;
       return (topo.dado);
    }
}
